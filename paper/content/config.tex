\renewenvironment{proof}[1][\proofname]{\par

  \normalfont \topsep0pt \partopsep0pt % Removes extra vertical space before proof

        \itshape
    #1\@addpunct{.}]\ignorespaces
}{%
  \popQED\endtrivlist\@endpefalse
}
\makeatother
